include <graphics.h> 
#include <conio.h> 
 main() 
{   int gd = DETECT, gm; 
    initgraph(&gd, &gm, "C:\\TC\\BGI"); 
    arc(320, 240, 0, 45, 100); 
    getch(); 
   closegraph();   return 0;    }

عبارة text whit diffrent fonts بشكل افقي خطوط مختلفة 

#include <graphics.h> 
#include <conio.h> 
main() 
{       int gd = DETECT, gm, x = 25, y = 25, font; 
    initgraph(&gd,&gm,"C:\\TC\\BGI"); 
    for (font = 0; font <= 10; font++) 
   { 
      settextstyle(font, HORIZ_DIR, font); 
      outtextxy(x, y, "Text with different fonts"); 
      y = y + 25; 
   } 
    getch(); 
   closegraph(); 
   return 0;  }
